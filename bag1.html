Perbedaan var, let, dan const

Lanjut ke Arrow Function →

Penjelasan Singkat

  • var: Dapat diubah dan dideklarasikan ulang, cakupan fungsi (function scope).
  • let: Dapat diubah, tidak bisa dideklarasikan ulang, cakupan blok (block scope).
  • const: Tidak bisa diubah maupun dideklarasikan ulang, cakupan blok (block scope).

Contoh Kode

var:

function contoh_var() {
  var a = 1;
  if (true) {
    var a = 2;
    console.log("di dalam if var " + a);
  }
  console.log("di luar if var " + a);
}

let:

function contoh_let() {
  let b = 1;
  if (true) {
    let b = 2;
    b = 5;
    console.log("di dalam if let " + b);
  }
  console.log("di luar if let " + b);
}

const:

function contoh_const() {
  const c = 1;
  console.log(c);
  // c = 2; // Akan error jika baris ini diaktifkan
}

const (error jika diubah):

function contoh_const_error() {
  const d = 10;
  d = 20; // Error: Assignment to constant variable
  console.log(d);
}
